1 # This is the client program that uses UDP
 2 
 3 # Sequence:
 4 #
 5 # 1. Create a socket
 6 # 2. Send messages to it
 7 # (We need to know the server's hostname and port.)
 8 
 9 import socket
10 
11 # create a socket
12 s=socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
13     # The first argument should be AF_INET
14     # The second argument is SOCK_STREAM for TCP service
15     #    and SOCK_DGRAM for UDP service
16 
17 # connect to the server
18 host='localhost'
19 port=43387  # this is the server's port number, which the client needs to know
20 
21 # send some bytes
22 s.sendto("I could tell you a joke but I'm not sure you'll get it.", (host,port))
23 
24 # close the connection
25 s.close()
