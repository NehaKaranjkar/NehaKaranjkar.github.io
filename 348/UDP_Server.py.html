1 # This is the Server program that uses UDP
 2 #
 3 # Sequence of steps:
 4 #   1. create a socket  
 5 #   2. bind the socket to a host and port
 6 #   3. start sending and receiving data on this socket
 7 
 8 
 9 import socket
10 
11 #  create a socket
12 s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
13     # The first argument should be AF_INET
14     # The second argument is SOCK_STREAM for TCP service
15     #    and SOCK_DGRAM for UDP service
16 
17 # bind it to a host and a port
18 host = 'localhost'
19 port = 43387  # arbitrarily chosen non-privileged port number
20 s.bind((host,port))
21 
22 print("Server started...waiting for a connection from the client")
23 
24 # receive some bytes and print them
25 data, addr = s.recvfrom(1024) # buffer size is 1024 bytes
26 print("MESSAGE=",data,"ADDR=",addr)
27 
28 # close the connection
29 s.close()
