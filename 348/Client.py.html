1 # This is the client program
 2 
 3 # Sequence:
 4 #
 5 # 1. Create a socket
 6 # 2. Connect it to the server process. 
 7 #   We need to know the server's hostname and port.
 8 # 3. Send and receive data 
 9 
10 import socket
11 
12 # create a socket
13 s=socket.socket(socket.AF_INET, socket.SOCK_STREAM)
14     # The first argument should be AF_INET
15     # The second argument is SOCK_STREAM for TCP service
16     #    and SOCK_DGRAM for UDP service
17 
18 # connect to the server
19 host='localhost'
20 port=43387  # this is the server's port number, which the client needs to know
21 s.connect((host,port))
22 
23 # send some bytes
24 s.send("Knock knock..")
25 
26 # read a response
27 response = s.recv(1024)
28 print(response)
29 
30 
31 # close the connection
32 s.close()
