1 # This is the Server program
 2 #
 3 # Sequence of steps:
 4 #   1. create a socket  
 5 #   2. bind the socket to a host and port
 6 #   3. start listening on this socket
 7 #   4. accept an incoming connection from the client
 8 #   5. send and receive data over the socket
 9 
10 
11 import socket
12 
13 #  create a socket
14 s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
15                 # use SOCK_STREAM for TCP
16                 # use SOCK_DGRAM for UDP
17 
18 # bind it to a host and a port
19 host = 'localhost'
20 port = 43387  # arbitrarily chosen non-privileged port number
21 s.bind((host,port))
22 
23 # start listening for TCP connections made to this socket
24 # the argument "1" is the max number of queued up clients allowed
25 s.listen(1)
26 print("Server started...waiting for a connection from the client")
27 
28 # accept a connection
29 c, addr = s.accept()
30 print("Connection initiated from ",addr)
31 
32 # receive some bytes and print them
33 # the argument 1024 is the maximum number of characters to be read at a time
34 response = c.recv(1024)
35 print(response)
36 
37 # send some bytes...
38 c.send("Whose there?")
39 
40 # close the connection
41 c.close()
